import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

// üîä Model class for a sound item
class SoundItem {
  final String label;
  final IconData icon;
  final Color color;

  const SoundItem(this.label, this.icon, this.color);
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Soundboard App',
      debugShowCheckedModeBanner: false, // removes debug banner
      home: const SoundboardScreen(),
    );
  }
}

class SoundboardScreen extends StatefulWidget {
  const SoundboardScreen({super.key});

  @override
  State<SoundboardScreen> createState() => _SoundboardScreenState();
}

class _SoundboardScreenState extends State<SoundboardScreen> {
  // üé∂ Initial list of sounds
  final List<SoundItem> sounds = [
    const SoundItem("Drum", Icons.music_note, Colors.blue),
    const SoundItem("Clap", Icons.pan_tool, Colors.green),
    const SoundItem("Bell", Icons.notifications, Colors.orange),
    const SoundItem("Guitar", Icons.queue_music, Colors.redAccent),
    const SoundItem("Piano", Icons.piano, Colors.teal),
    const SoundItem("Whistle", Icons.speaker, Colors.pink),
  ];

  // üéµ Function to add new sound dynamically
  void _addSound() {
    setState(() {
      sounds.add(
        SoundItem("New Sound ${sounds.length + 1}", Icons.music_note, Colors.purple),
      );
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("Soundboard Application"),
        backgroundColor: Colors.deepPurple,
      ),
      body: Container(
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [
            const Text(
              "üéµ Welcome to Soundboard!",
              style: TextStyle(color: Colors.black, fontSize: 18),
            ),
            const SizedBox(height: 16),

            // üîç Search Bar
            const TextField(
              decoration: InputDecoration(
                hintText: "Search Sound",
                prefixIcon: Icon(Icons.search),
                border: OutlineInputBorder(),
              ),
            ),
            const SizedBox(height: 16),

            // üîä Dynamic ListView (scrollable)
            Expanded(
              child: ListView.builder(
                itemCount: sounds.length,
                itemBuilder: (context, index) {
                  final sound = sounds[index];
                  return soundCard(sound);
                },
              ),
            ),

            const SizedBox(height: 16),

            // üé∂ Visualizer Placeholder
            Container(
              height: 120,
              width: double.infinity,
              decoration: BoxDecoration(
                color: Colors.deepPurple.shade100,
                borderRadius: BorderRadius.circular(12),
              ),
              child: const Center(
                child: Text(
                  "üé∂ Sound Visualizer Area",
                  style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold),
                ),
              ),
            ),
          ],
        ),
      ),

      // ‚ûï Floating Button to add new sound dynamically
      floatingActionButton: FloatingActionButton(
        onPressed: _addSound,
        backgroundColor: Colors.deepPurple,
        child: const Icon(Icons.add),
      ),
    );
  }

  /// Custom ListTile-style sound card
  Widget soundCard(SoundItem sound) {
    return Card(
      margin: const EdgeInsets.symmetric(vertical: 6),
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
      elevation: 3,
      child: ListTile(
        leading: CircleAvatar(
          backgroundColor: sound.color,
          child: Icon(sound.icon, color: Colors.white),
        ),
        title: Text(
          sound.label,
          style: const TextStyle(fontSize: 16, fontWeight: FontWeight.w500),
        ),
        trailing: ElevatedButton(
          onPressed: () {
            debugPrint("Playing ${sound.label} sound...");
          },
          style: ElevatedButton.styleFrom(
            backgroundColor: sound.color,
            foregroundColor: Colors.white,
            shape: RoundedRectangleBorder(
                borderRadius: BorderRadius.circular(12)),
          ),
          child: const Text("Play"),
        ),
      ),
    );
  }
}
